JS

参数x为当前时间占比

function easeOutElastic(x: number): number {
      const c4 = (2 * Math.PI) / 3;
      
      return x === 0
        ? 0
        : x === 1
        ? 1
        : pow(2, -10 * x) * sin((x * 10 - 0.75) * c4) + 1;
      }
