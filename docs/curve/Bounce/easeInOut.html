JS

参数x为当前时间占比

function easeInOutBounce(x: number): number {
      return x < 0.5
        ? (1 - easeOutBounce(1 - 2 * x)) / 2
        : (1 + easeOutBounce(2 * x - 1)) / 2;
      }
